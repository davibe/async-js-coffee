ES6 Generators

Q = require 'q'
client = require './client_promises'


Q.spawn ->
  try
    yield client.login('davide', 'pass')
    posts = yield client.getPosts('alberto')
    results = []
    for post of posts
      results.push(yield client.countPostLikes(post))
    yield client.logout()
  catch e
    console.error 'should handle error here ..'
